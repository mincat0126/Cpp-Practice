# 멤버 연산자 함수
&nbsp;&nbsp;&nbsp;**선언**: 리턴타입 Class::operator 연산자(인수 목록){내용;}  
> 연산자 자리에는 +, -, *, /, <<, != 등 대부분의 연산자 기호가 올 수 있다.  
> 연산자 함수의 인수란 피연산자를 의미한다.  

<pre>
#include < iostream >
  
class Complex
{
private:
     double real;
     double image;

public:
     Complex() { }
     Complex(double r, double i) : real(r), image(i) { }
     void OutComplex() const { printf("%.2f+%.2fi\n",real,image); }
    const Complex operator +(const Complex &T) const {
        Complex R;
        R.image = image + T.image;
        R.real = real + T.real;
        return R;
    }
};
