# 큐
**큐**: FIFO 원리로 동작하는 자료구조

<pre>#include < stdio.h >
#include < windows.h >

int* Queue;
int QSize;
int head, tail;

void InitQueue(int size)
{
    QSize = size;
    Queue = (int*)malloc(QSize * sizeof(int));
    head = tail = 0;
}

void FreeQueue()
{
    free(Queue);
}

BOOL Insert(int data)
{
    if ((tail + 1) % QSize == head)
    {
        return FALSE;
    }

    Queue[tail] = data;
    tail = (tail + 1) % QSize;
    return TRUE;
}

int Delete()
{
    int data;

    if (head == tail)
    {
        return -1;
    }

    data = Queue[head];
    head = (head + 1) % QSize;
    return data;

}

void main()
{
    int i;
    InitQueue(10);

    printf("빈 상태에서 삭제할 때 = %d\n", Delete());

    for (i = 0; i < 9; i++)
    {
        Insert(i);
    }
    printf("가득찬 상태에서 삽입 %s\n", Insert(100) ? "성공" : "실패");

    for (i = 0; i < 9; i++)
    {
        printf("%d  ", Delete());
    }

    FreeQueue();
}</pre>
